<Section icon={Target} title="Fortschritt und Struktur der Kampagne">

<Steps>
  <Step num="1" title="Phase Eins: Verwirrung">

  </Step>
  <Step num="2" title="Phase Zwei: Neuformierung">
    Verbinde dich mit den Überresten des Militärs und bilde den Kern des Trupps, während das Ausmaß der Invasion deutlich wird.

  <Step num="3" title="Phase Drei: Verzweiflung">
